CSS Fundamentals

Colors & Backgrounds

/* Named colors */
.color1 { background-color: tomato; }

/* Hex colors */
.color2 { background-color: #4361ee; }

/* RGB colors */
.color3 { background-color: rgb(114, 9, 183); }

/* RGBA colors */
.color4 { background-color: rgba(67, 97, 238, 0.5); }

/* Gradients */
.gradient {
  background: linear-gradient(90deg, #4361ee, #7209b7);
}
